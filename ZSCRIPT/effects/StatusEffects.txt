Class StatusEffect : Actor
{
	int y_offset;
	
	default
	{
		+NoClip;
	}
	
	override void tick() {
	
		if (target) {
		
			if (target.health <= 0) {
				Self.destroy();
			} else {
				Vector3 follow = target.pos;
				SetXYZ(follow);
			}

		} else {
			Self.destroy();
		}
		
		Super.tick();
	}

	states
	{
		Spawn:
			stop;
		Frozen:
			SICE A 1 Bright;
			loop;
		Burned:
			SBUR ABCDEFGHIJKLMNOPQRSTU 1 Bright;
			goto Waiting;
		Paralyzed:
			SPAR ABABABABABABABABABABABABABABABAB 1 Bright;
			goto Waiting;
		Confused:
		
		Waiting:
			TNT1 A 1;
			loop;
	}
}
