Class RageCandyBar : CustomInventory
{
	default
	{
		XScale 0.15;
		YScale 0.15;
		+Floatbob;
		Inventory.PickupMessage "Ate a Rage Candy Bar! Max anger raised!";
	}

	states
	{
		Spawn:
			RAGE A 1;
			Loop;
		Pickup:
			TNT1 A 0
			{
				if (Gardevoir(players[consoleplayer].mo).RageConstant < 1000)
				{
					Gardevoir(players[consoleplayer].mo).IncreaseAngerRange(50);
				} 
				else 
				{
					Gardevoir(players[consoleplayer].mo).IncreaseAngerRange(25);
				}
			}
			stop;
	}
}
