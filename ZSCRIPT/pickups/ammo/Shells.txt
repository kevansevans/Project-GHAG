Class GHAGShell : CustomInventory replaces Shell
{
	default {
		Inventory.PickupMessage "$GOTSHELLS";
		Inventory.pickupsound "misc/w_pkup";
		Inventory.amount 4;
	}
	
	override bool TryPickup(Actor toucher)
	{
		let gardie = Gardevoir(Players[consoleplayer].mo);
		for (let probe = gardie.Inv; probe != NULL; probe = probe.Inv)
		{
			let item = Shell(probe);
			if (item) {
				if (gardie.countinv('Shell') >= item.maxamount) return false;
				else {
					if (gardie.countinv("GHAGShotgun") == 0)
					{
						gardie.giveInventory("GHAGShotgun", 1);
						gardie.giveInventory("GHAGShotgunOverlay", 1);
						if (!gardie.chargingBlackHole) gardie.A_SelectWeapon("GHAGShotgun");
					}
					gardie.giveInventory("Shell", self.amount);
				}
			}
		}
		GoAwayAndDie();
		return true;
	}
	
	States
	{
		Spawn:
			SHEL A -1;
			Stop;
	}
}
