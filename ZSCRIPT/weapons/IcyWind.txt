Class IcyWind : PokemonMove
{
	Default
	{
		Weapon.SlotNumber 5;
		
		Tag "Icy Wind";
		Inventory.PickupMessage "Tada! Gardevoir learned \cnIcy Wind!"; // and thankfully didn't forget anything!
	}
	
	States
	{
		Spawn:
			TMIC A -1;
			stop;
		Select:
			HAND A 1 A_GHAGRaise();
			Loop;
		Deselect:
			HAND A 1 A_GHAGLower();
			Loop;
		Ready:
			HAND A 1 A_WeaponReady();
			Loop;
		Fire:
			TNT1 A 0 A_FireProjectile("IcyWindProjectile");
			HAND ABCDE 1;
			TNT1 A 0 A_Refire();
			goto Ready;
	}
}

Class IcyWindProjectile : Actor
{
	Default
	{
		Projectile;
		DamageType "Ice";
		+Ripper;
		Damage 1;
		Speed 5;
	}
	States
	{
		Spawn:
			ICYA AB 4 Bright;
			ICYA CDDEFGGHIJKLMNOPQRSTUVWXYZ 1 Bright;
			ICYB ABCDEFGHIJKLMNOPQ 1 Bright;
			TNT1 A 1;
			ICYB RSTTUV 1 Bright;
			TNT1 A 2;
			ICYB V 5;
			Loop;
		Death:
			TNT1 A 1;
			Stop;
	}
}
