{
	states 
	{
		Pickup:
			TNT1 A 0 {
				
				for (int i; i <= 3; ++i) {
					let sphere = Soulsphere(self.spawn("SoulSPhere"));

				A_GiveInventory("BlueArmor");
			}
	}
}
